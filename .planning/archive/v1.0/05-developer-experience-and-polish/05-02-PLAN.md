3. Update `crates/textual-rs/src/widget/context.rs`:
   - Add field: `pub worker_tx: Option<flume::Sender<(WidgetId, Box<dyn std::any::Any + Send>)>>`
   - Add field: `pub worker_handles: std::cell::RefCell<slotmap::SecondaryMap<WidgetId, Vec<tokio::task::AbortHandle>>>`
   - Initialize both in `AppContext::new()`: `worker_tx: None`, `worker_handles: RefCell::new(SecondaryMap::new())`
   - Add method (per D-05):
   ```rust
   /// Spawn an async worker tied to a widget. The worker runs on the Tokio LocalSet.
   /// On completion, the result is delivered as a WorkerResult<T> message to the
   /// source widget's message queue entry. T must be Send + 'static.
   /// Returns an AbortHandle for manual cancellation.
   pub fn run_worker<T: Send + 'static>(
       &self,
       source_id: WidgetId,
       fut: impl std::future::Future<Output = T> + 'static,
   ) -> tokio::task::AbortHandle {
       let tx = self.worker_tx.clone().expect("worker_tx not initialized — run_worker called outside App::run()");
       let handle = tokio::task::spawn_local(async move {
           let result = fut.await;
           let _ = tx.send((source_id, Box::new(crate::worker::WorkerResult { value: result })));
       });
       let abort = handle.abort_handle();
       // Track handle for auto-cancel on unmount
       let mut handles = self.worker_handles.borrow_mut();
       handles.entry(source_id).unwrap().or_insert_with(Vec::new).push(abort.clone());
       abort
   }
   ```
   - Add method for unmount cleanup:
   ```rust
   /// Cancel all workers associated with a widget. Called during unmount.
   pub fn cancel_workers(&self, widget_id: WidgetId) {
       if let Some(handles) = self.worker_handles.borrow_mut().remove(widget_id) {
           for handle in handles {
               handle.abort();
           }
       }
   }
   ```
